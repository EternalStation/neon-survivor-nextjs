## Target Implementation

```typescript
// update/EnemyUpdateDispatch.ts (~50 lines)
import { updateNormalEnemy } from './NormalEnemyUpdate'
import { updateEliteEnemy } from './EliteEnemyUpdate'
import { updateZombie } from '../unique/ZombieLogic'
// ... etc

export function updateEnemy(enemy: Enemy, state: GameState, step: number) {
  if (enemy.worm) return updateWorm(enemy, state, step)
  if (enemy.boss) return updateBoss(enemy, state, step)
  if (enemy.elite?.isElite) return updateEliteEnemy(enemy, state, step)

  switch (enemy.type) {
    case 'zombie': return updateZombie(enemy, state, step)
    case 'snitch': return updateSnitch(enemy, state, step)
    case 'glitcher': return updateGlitcher(enemy, state, step)
    case 'prism': return updatePrism(enemy, state, step)
    default: return updateNormalEnemy(enemy, state, step)
  }
}
```
